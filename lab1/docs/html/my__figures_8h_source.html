304  std::vector<Point> Figure::intersectionOfCircleAndLine(Figure circle, Figure line)
305  {
306  if (circle.type == FiguresType::LINE && line.type == FiguresType::CIRCLE)
307  return intersectionOfCircleAndLine(line, circle);
308  if (circle.type != FiguresType::CIRCLE || line.type != FiguresType::LINE)
309  return {};
310  double distance = findDistanceFromLineToPoint(line, circle.first);
311  double radius = distanceBetweenPoints(circle.first, circle.second);
312  if (distance > radius) return {};
313  Equation equation1{ line };
314  //Py^2 + Qy + D = 0
315  double coefficientP = 0, coefficientQ = 0, coefficientD = 0;
316  if (equation1.a == 0)
317  {
318  coefficientP = 1;
319  coefficientD = pow(circle.first.x, 2) + pow(equation1.c / equation1.b - circle.first.y, 2) - pow(radius, 2);
320  coefficientQ = -2 * circle.first.x;
321  }
322  else
323  {
324  coefficientP = pow(equation1.b / equation1.a, 2) + 1;
325  coefficientD = pow((equation1.c - equation1.a * circle.first.x) / equation1.a, 2)
326  + pow(circle.first.y,2) - pow(radius, 2);
327  coefficientQ = -2 * equation1.b * (equation1.c - equation1.a * circle.first.x) / pow(equation1.a, 2) - 2 * circle.first.y;
328  }
329  std::vector<double> y = solveQuadraticEquation(coefficientP, coefficientQ, coefficientD);
330  switch (y.size())
331  {
332  case 0: return {};
333  break;
334  case 1: if(equation1.a != 0) return { { equation1.c / equation1.a - (equation1.b / equation1.a) * y[0], y[0]} };
335  else return { { y[0], equation1.c / equation1.b} };
336  break;
337  case 2: if (equation1.a != 0) return { {equation1.c / equation1.a - (equation1.b / equation1.a) * y[0], y[0]}
338  ,{ equation1.c / equation1.a - (equation1.b / equation1.a) * y[1], y[1]} };
339  else return { { y[0], equation1.c / equation1.b}
340  ,{ y[1], equation1.c / equation1.b } };
341  break;
342  }
343  return {};
344  }

351  std::vector<Point> Figure::intersectionOfTwoCirclesByCoordinates(double x1, double x2, double y1, double y2, double radius1, double radius2)
352  {
353  //x = B*y + C
354  double coefficientB = -1 * (y2 - y1) / (x2 - x1);
355  double coefficientC = (pow(radius1, 2) - pow(radius2, 2) - pow(x1, 2)
356  + pow(x2, 2) - pow(y1, 2) + pow(y2, 2))
357  / (2 * (x2 - x1));
358  //P*y^2 + Q*y + D = 0
359  double coefficientP = pow(coefficientB, 2) + 1;
360  double coefficientQ = 2 * (coefficientB * (coefficientC - x1) - y1);
361  double coefficientD = pow(coefficientC - x1, 2) + pow(y1, 2) - pow(radius1, 2);
362  std::vector<double> y = solveQuadraticEquation(coefficientP, coefficientQ, coefficientD);
363  switch (y.size())
364  {
365  case 0: return {};
366  break;
367  case 1: return { { coefficientB * y[0] + coefficientC, y[0]} };
368  break;
369  case 2: return { {coefficientB * y[0] + coefficientC, y[0]}
370  ,{ coefficientB * y[1] + coefficientC, y[1]} };
371  break;
372  }
373 
374  return {};
375  }

422  Point Figure::inversionTransformationOfPointByCircle(Figure circle, Point point)
423  {
424  if (circle.type != FiguresType::CIRCLE) return point;
425  constexpr double max = std::numeric_limits<double>::max();
426  constexpr double min = std::numeric_limits<double>::min();
427  if (point == circle.first) return Point(max, max);
428  if ((point.x == doctest::Approx(max) || point.x == doctest::Approx(min))
429  && (point.y == doctest::Approx(max) || point.y == doctest::Approx(min)))
430  return circle.first;
431  double x0 = circle.first.x, y0 = circle.first.y;
432  double radius = distanceBetweenPoints(circle.first, circle.second);
433  double divisor = (pow(point.x - x0, 2) + pow(point.y - y0, 2));
434  double x = x0 + (pow(radius, 2) * (point.x - x0)) / divisor;
435  double y = y0 + (pow(radius, 2) * (point.y - y0)) / divisor;
436 
437  return Point(x, y);
